# Kotlin 设计模式与最佳实践

## 一、创建型模式

### 1. 单例模式（Singleton）

```kotlin
// 方式1：object 声明（推荐，线程安全，懒加载）
object AppConfig {
    var apiUrl = "https://api.example.com"
    fun init(url: String) { apiUrl = url }
}

// 方式2：带参数的单例（双重检查锁）
class DatabaseHelper private constructor(context: Context) {
    companion object {
        @Volatile
        private var INSTANCE: DatabaseHelper? = null

        fun getInstance(context: Context): DatabaseHelper {
            return INSTANCE ?: synchronized(this) {
                INSTANCE ?: DatabaseHelper(context.applicationContext).also { INSTANCE = it }
            }
        }
    }
}

// 方式3：by lazy（线程安全）
class Preferences(context: Context) {
    companion object {
        private lateinit var appContext: Context
        val instance: Preferences by lazy { Preferences(appContext) }
        fun init(context: Context) { appContext = context.applicationContext }
    }
}
```

### 2. 工厂模式

```kotlin
// 简单工厂（companion object）
sealed class Notification {
    data class Email(val address: String) : Notification()
    data class SMS(val phone: String) : Notification()
    data class Push(val token: String) : Notification()

    companion object {
        fun create(type: String, target: String): Notification = when (type) {
            "email" -> Email(target)
            "sms" -> SMS(target)
            "push" -> Push(target)
            else -> throw IllegalArgumentException("Unknown type: $type")
        }
    }
}

// 抽象工厂
interface ViewModelFactory<T : ViewModel> {
    fun create(): T
}

class HomeViewModelFactory(
    private val repo: HomeRepository
) : ViewModelFactory<HomeViewModel> {
    override fun create() = HomeViewModel(repo)
}
```

### 3. Builder 模式

```kotlin
// Kotlin 风格：使用默认参数 + apply 替代传统 Builder
data class NetworkConfig(
    val baseUrl: String,
    val timeout: Long = 30_000,
    val retryCount: Int = 3,
    val headers: Map<String, String> = emptyMap(),
    val interceptors: List<Interceptor> = emptyList()
)

// 使用
val config = NetworkConfig(
    baseUrl = "https://api.example.com",
    timeout = 10_000,
    headers = mapOf("Authorization" to "Bearer xxx")
)

// 复杂场景仍可使用 Builder + DSL
class Dialog private constructor(
    val title: String,
    val message: String,
    val positiveText: String?,
    val onPositive: (() -> Unit)?
) {
    class Builder {
        var title: String = ""
        var message: String = ""
        var positiveText: String? = null
        var onPositive: (() -> Unit)? = null

        fun build() = Dialog(title, message, positiveText, onPositive)
    }
}

fun dialog(init: Dialog.Builder.() -> Unit): Dialog {
    return Dialog.Builder().apply(init).build()
}

// DSL 风格使用
val myDialog = dialog {
    title = "提示"
    message = "确定删除？"
    positiveText = "确定"
    onPositive = { deleteItem() }
}
```

## 二、结构型模式

### 1. 装饰器模式（委托实现）

> **面试加分项**：Kotlin `by` 关键字实现委托

```kotlin
interface DataSource {
    fun readData(): String
    fun writeData(data: String)
}

class FileDataSource(private val filename: String) : DataSource {
    override fun readData() = File(filename).readText()
    override fun writeData(data: String) = File(filename).writeText(data)
}

// 使用 by 委托 + 装饰
class LoggingDataSource(
    private val wrapped: DataSource
) : DataSource by wrapped {
    override fun writeData(data: String) {
        println("Writing data: ${data.take(20)}...")
        wrapped.writeData(data)
        println("Write completed")
    }
    // readData 自动委托给 wrapped
}

class EncryptingDataSource(
    private val wrapped: DataSource
) : DataSource by wrapped {
    override fun readData() = decrypt(wrapped.readData())
    override fun writeData(data: String) = wrapped.writeData(encrypt(data))
}

// 组合使用
val dataSource = LoggingDataSource(EncryptingDataSource(FileDataSource("data.txt")))
```

## 三、行为型模式

### 1. 观察者模式

```kotlin
// Kotlin 原生实现
class EventBus {
    private val listeners = mutableMapOf<String, MutableList<(Any) -> Unit>>()

    fun subscribe(event: String, listener: (Any) -> Unit) {
        listeners.getOrPut(event) { mutableListOf() }.add(listener)
    }

    fun publish(event: String, data: Any) {
        listeners[event]?.forEach { it(data) }
    }
}

// Android 中：LiveData / StateFlow 就是观察者模式的实现
// Flow 版本
class UserRepository {
    private val _users = MutableSharedFlow<List<User>>()
    val users: SharedFlow<List<User>> = _users.asSharedFlow()

    suspend fun refresh() {
        val data = api.fetchUsers()
        _users.emit(data)
    }
}
```

### 2. 策略模式

```kotlin
// 使用高阶函数实现（更 Kotlin 化）
class Sorter<T>(private val strategy: (List<T>) -> List<T>) {
    fun sort(list: List<T>) = strategy(list)
}

val alphabetical = Sorter<String> { it.sorted() }
val byLength = Sorter<String> { it.sortedBy { s -> s.length } }
val reversed = Sorter<String> { it.sortedDescending() }

// 实际应用：图片压缩策略
typealias CompressStrategy = (Bitmap, Int) -> ByteArray

val jpegCompress: CompressStrategy = { bitmap, quality ->
    ByteArrayOutputStream().also {
        bitmap.compress(Bitmap.CompressFormat.JPEG, quality, it)
    }.toByteArray()
}

val webpCompress: CompressStrategy = { bitmap, quality ->
    ByteArrayOutputStream().also {
        bitmap.compress(Bitmap.CompressFormat.WEBP, quality, it)
    }.toByteArray()
}
```

### 1. 作用域控制

### 2. 协程最佳实践

```kotlin
// 使用 Result 包装可失败操作
suspend fun safeApiCall(block: suspend () -> Response): Result<Response> {
    return try {
        Result.success(block())
    } catch (e: Exception) {
        Result.failure(e)
    }
}

// 使用 callbackFlow 将回调转为 Flow
fun locationUpdates(): Flow<Location> = callbackFlow {
    val listener = LocationListener { location ->
        trySend(location)
    }
    locationManager.requestUpdates(listener)
    awaitClose { locationManager.removeUpdates(listener) }
}
```

### 3. 契约（Contracts）

```kotlin
@OptIn(ExperimentalContracts::class)
fun String?.isNotNullOrEmpty(): Boolean {
    contract {
        returns(true) implies (this@isNotNullOrEmpty != null)
    }
    return this != null && isNotEmpty()
}

// 使用后编译器知道非空
val name: String? = getName()
if (name.isNotNullOrEmpty()) {
    println(name.length)  // 智能转换为 String（非空）
}
```
