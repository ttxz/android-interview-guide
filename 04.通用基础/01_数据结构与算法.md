## 一、核心数据结构

### 1. 数组与链表

| 操作 | 数组 | 链表 |
|------|------|------|
| 随机访问 | O(1) | O(n) |
| 头部插入 | O(n) | O(1) |

**面试常问**：ArrayList vs LinkedList

- ArrayList：基于动态数组，随机访问快，扩容为 1.5 倍
- LinkedList：基于双向链表，插入删除快，内存不连续

fun isValidBrackets(s: String): Boolean {
    val stack = ArrayDeque<Char>()
    val map = mapOf(')' to '(', ']' to '[', '}' to '{')
    for (c in s) {
        if (c in "({[") stack.push(c)
        else if (c in ")}]") {
            if (stack.isEmpty() || stack.pop() != map[c]) return false
        }
    }
    return stack.isEmpty()
}

## 二、排序算法

| 算法 | 平均 | 最差 | 空间 | 稳定 |
|------|------|------|------|------|
| 冒泡 | O(n²) | O(n²) | O(1) | ✅ |
| 选择 | O(n²) | O(n²) | O(1) | ❌ |
| 插入 | O(n²) | O(n²) | O(1) | ✅ |
| **快排** | **O(nlogn)** | O(n²) | O(logn) | ❌ |
| **归并** | **O(nlogn)** | O(nlogn) | O(n) | ✅ |
| 堆排 | O(nlogn) | O(nlogn) | O(1) | ❌ |

```kotlin
// 快速排序
fun quickSort(arr: IntArray, low: Int = 0, high: Int = arr.size - 1) {
    if (low >= high) return
    val pivot = partition(arr, low, high)
    quickSort(arr, low, pivot - 1)
    quickSort(arr, pivot + 1, high)
}

fun partition(arr: IntArray, low: Int, high: Int): Int {
    val pivot = arr[high]
    var i = low
    for (j in low until high) {
        if (arr[j] < pivot) {
            arr[i] = arr[j].also { arr[j] = arr[i] }
            i++
        }
    }
    arr[i] = arr[high].also { arr[high] = arr[i] }
    return i
}
```

## 三、常见面试算法题型

### 1. 双指针

```kotlin
// 两数之和（有序数组）
fun twoSum(nums: IntArray, target: Int): IntArray {
    var left = 0; var right = nums.size - 1
    while (left < right) {
        val sum = nums[left] + nums[right]
        when {
            sum == target -> return intArrayOf(left, right)
            sum < target -> left++
            else -> right--
        }
    }
    return intArrayOf()
}
```

### 2. 滑动窗口

```kotlin
// 最长无重复子串
fun lengthOfLongestSubstring(s: String): Int {
    val set = mutableSetOf<Char>()
    var left = 0; var maxLen = 0
    for (right in s.indices) {
        while (s[right] in set) set.remove(s[left++])
        set.add(s[right])
        maxLen = maxOf(maxLen, right - left + 1)
    }
    return maxLen
}
```

### 3. 二分查找

```kotlin
fun binarySearch(nums: IntArray, target: Int): Int {
    var left = 0; var right = nums.size - 1
    while (left <= right) {
        val mid = left + (right - left) / 2
        when {
            nums[mid] == target -> return mid
            nums[mid] < target -> left = mid + 1
            else -> right = mid - 1
        }
    }
    return -1
}
```

### 4. BFS / DFS

```kotlin
// BFS 层序遍历
fun levelOrder(root: TreeNode?): List<List<Int>> {
    root ?: return emptyList()
    val result = mutableListOf<List<Int>>()
    val queue: Queue<TreeNode> = LinkedList()
    queue.offer(root)
    while (queue.isNotEmpty()) {
        val level = mutableListOf<Int>()
        repeat(queue.size) {
            val node = queue.poll()
            level.add(node.`val`)
            node.left?.let { queue.offer(it) }
            node.right?.let { queue.offer(it) }
        }
        result.add(level)
    }
    return result
}
```

### 5. 动态规划

```kotlin
// 最长公共子序列
fun lcs(text1: String, text2: String): Int {
    val m = text1.length; val n = text2.length
    val dp = Array(m + 1) { IntArray(n + 1) }
    for (i in 1..m) {
        for (j in 1..n) {
            dp[i][j] = if (text1[i - 1] == text2[j - 1])
                dp[i - 1][j - 1] + 1
            else
                maxOf(dp[i - 1][j], dp[i][j - 1])
        }
    }
    return dp[m][n]
}
```

|--------|------------|
| n ≤ 20 | O(2^n) / O(n!) |
| n ≤ 100 | O(n³) |
| n ≤ 1000 | O(n²) |
| n ≤ 10⁶ | O(nlogn) |
| n ≤ 10⁸ | O(n) |
| n > 10⁸ | O(logn) / O(1) |
